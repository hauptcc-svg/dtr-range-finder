### Step 5: Create Keep-Alive Script

Create new file: `keep_alive.py`

```python
"""
Keep Replit Running 24/7
Prevents Replit from sleeping due to inactivity
"""

from flask import Flask
import threading
import time
from urllib import request
import os

app = Flask(__name__)

@app.route('/')
def home():
    return "✅ DTR Trading Agent is Running"

@app.route('/health')
def health():
    return {"status": "healthy", "timestamp": time.time()}

def run_flask():
    """Run Flask server on port 8080"""
    try:
        app.run(host='0.0.0.0', port=8080, debug=False)
    except Exception as e:
        print(f"Flask error: {e}")

def keep_alive():
    """Ping the Replit URL every 5 minutes to prevent sleeping"""
    replit_url = os.getenv("REPLIT_URL", "https://dtr-trading-agent.repl.co")
    
    while True:
        try:
            request.urlopen(replit_url, timeout=5)
            print(f"✓ Keep-alive ping sent to {replit_url}")
        except Exception as e:
            print(f"Keep-alive error: {e}")
        
        time.sleep(300)  # Ping every 5 minutes

def start_background_threads():
    """Start Flask and keep-alive in background threads"""
    
    # Start Flask server
    flask_thread = threading.Thread(target=run_flask, daemon=True)
    flask_thread.start()
    print("✓ Flask server started on port 8080")
    
    # Start keep-alive pinger
    keep_alive_thread = threading.Thread(target=keep_alive, daemon=True)
    keep_alive_thread.start()
    print("✓ Keep-alive thread started")

if __name__ == "__main__":
    start_background_threads()
    
    # Keep main thread alive
    try:
        while True:
            time.sleep(1)
    except KeyboardInterrupt:
        print("Keep-alive script stopped")
```
